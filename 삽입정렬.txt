//내가 한 거

#include <stdio.h>	
int main() {
	int arr[10] = { 3, 8, 9, 10, 5, 7, 4, 1, 2, 6 };
	int temp;

	for (int i = 1; i < 10; i++) {
		for (int j = 0; j < i; j++) {
			if (arr[j] > arr[i]) {
				temp = arr[i];
				for (int k = i; k > j; k--) {
					arr[k] = arr[k - 1];
				}
				arr[j] = temp;
			}
		}
		for (int i = 0; i < 10; i++) {
			printf("%d ", arr[i]);
	}

}
